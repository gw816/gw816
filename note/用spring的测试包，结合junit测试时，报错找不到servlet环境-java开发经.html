解决方法：

除了在类名上方加入
@RunWith(SpringJUnit4ClassRunner.class)
@ContextConfiguration(locations = { "classpath*:/servlet-context.xml" })

外，还得加入@WebAppConfiguration

正确配置：

  1. @WebAppConfiguration
  2. @RunWith(SpringJUnit4ClassRunner.class)
  3. @ContextConfiguration(locations = { "classpath*:/servlet-context.xml" })
  4. public class MyTestDemo {
  5.        
  6.         //private static final Logger logger = Logger.getLogger(TestMybatis.class);
  7.         @Autowired
  8.         IUserInfoService userInfoService;
  9.         public IUserInfoService getUserService() {
  10.                 return userInfoService;
  11.         }
  12.         public void setUserService(IUserInfoService userService) {
  13.                 this.userInfoService = userService;
  14.         }


  17.         @Test
  18.                 public void testUser(){
  19.                 UserInfo u=userInfoService.getUserInfo(1);
  20.                 System.out.println(u.getNickname());
  21.                  }
  22. }
